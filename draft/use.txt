


meInit();

vp = medea.CreateViewport('glcanvas');
mesh = medea.CreateAsset(medea.Resource('teaport.asset'));

root = medea.GetRoot();
node = medea.CreateNode("a child");
node.AddEntity(mesh);
root.AddChild(node);

animspeed = 0.1;
node.AddAnimator(function(node,time,dtime) {
	node.RotateX(dtime*animspeed);
});
